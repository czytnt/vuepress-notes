    # Vue

    Author-Allen

    Time-2022/11/16

    # 1. 模板语法

    # 1.1 插值语法

        <h1>插值语法</h1>
    <h3>{{name}}</h3>   多用于标签体内的值
    // {{xxx}}   xxx 是 js 表达式，且可以直接读取到data中的所有属性

    # 1.2 指令语法

        <a v-bind:href="url">链接</a>
    <a v-bind:href="url" :x="hello">链接</a>
    <!-- v-bind 动态的绑定一个值，可以简写成 ： -->

        //指令语法的多级结构
        //若有重名的data属性值，则可以写为
        new Vue({ 
            el:'#xxid',
            data:{
                name:'hello',
                url:'http://cn.bing.com'
                school:{
                    name:'你好',
                    url:'http://baidu.com'
                }
            }
        })
        <a v-bind:href="url" :x="hello">{{ school.name }}</a>

    # 2. 数据绑定

    # 2.1 单向与双向数据绑定

        <div id="root">
            <!-- 单向 v-bind ，即只能从控制台改变数据影响页面上数据，data 流向页面，不能反向影响 -->
            <!-- 双向 v-model ，即控制台和页面可双向影响 -->
            单向数据绑定：<input type="text" v-bind:value="xname"> <br>
            双向数据绑定：<input type="text" v-model:value="xname"> <br>

            <!-- 简写
            
                单向数据绑定：<input type="text" :value="xname"> <br>
                双向数据绑定：<input type="text" v-model="xname"> <br>
            
             -->

            <!-- 如下代码错误，因为 v-model 只能应用在表单类元素（输入类元素） -->
            <h3 v-model:v-text:="xname">hello</h3>   //v-model:value 可简写为v-model:
        </div> 
        <script type="text/javascript">
            new Vue({
                el:"#root",
                data:{
                    xname:'你好'
                }
            })
        </script>

    # 3. el与data的两种写法

    # 3.1 el的常规写法（new的时候就知道给哪个容器服务）

       <script type="text/javascript">
            new Vue({
                el:"#root",
                data:{
                    xname:'你好'
                }
            })
        </script>

    # 3.2 回头再指定el服务对象

        <script type="text/javascript">
            const v=new Vue({
                data:{
                    xname:'你好'
                }
            })

            // console.log(v);

            setTimeout(()=>{
                v.$mount('#root')   //用这个方法，等1s钟之后才显示data数据 ==>一开始只显示{{name}}
            },1000); 

        </script>

    # 3.3 data对象式写法

        data:{
                    xname:'你好'
                }

    # 3.4 data的函数式写法

        data:function(){
        return {
            name:'Hello'
        }
    }

    # 4.MVVM模型

    <!-- 
    MVVM模型：
        1.M：模型（Moudle）：在data中的数据
        2.V：视图（View）：模板代码
        3.VM：视图模型（ViewMoudle）：Vue实例
    1.data中所有的属性，最后都出现在vm（Vue实例）上
    2.vm上所有的属性以及Vue原型上的所有属性，在Vue模板中都可以直接使用
-->

<div id="root">
        <!-- 准备好一个容器 -->
        <h1>学校名称：{{name}}</h1>   --->对应 View
        <h1>学校地址：{{address}}</h1>
    </div>
    <script type="text/javascript">
        Vue.config.productionTip = false //阻止Vue在启动时生成生产提示
        new Vue({                   --->对应 ViewMoudle
            el: '#root',
            data: {                 --->对应Model
                name: 'tnt',
                address: '重庆'
            }
        })
    </script>
